\subsubsection{Spherical coordinates}
\hskip-0.5cm
\begin{minipage}[c]{0.55\linewidth}
\begin{tabular}{>{$\displaystyle}l<{$}}
	x = r\sin\theta\cos\phi\\
	y = r\sin\theta\sin\phi\\
	z = r\cos\theta\\
	r = \sqrt{x^2+y^2+z^2}\\
	\theta = \textrm{acos}\frac{z}{\sqrt{x^2+y^2+z^2}}\\
	\phi = \textrm{atan2}(y,x)
\end{tabular}
\end{minipage}
